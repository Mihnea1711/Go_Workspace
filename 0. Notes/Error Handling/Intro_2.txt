package main

import (
	"errors"
	"fmt"
	"log"
	"os"
)

//errors is s standard package to handle errors
//os has some built in error types (look at docs)
//errors.Is checks the errors correctly

func main() {
	//this is a 2 part if stmt
	//the ; separates the init component from the test component
	if _, err := os.Open("myfile.txt"); err != nil {
		//we try to open the file and if there is an error this block gets executed
		if errors.Is(err, os.ErrNotExist) {
			log.Println("File does not exist!")
		} else {
			log.Println(err)
		}
		return
	}
	fmt.Println("File successfully opened!")
}
