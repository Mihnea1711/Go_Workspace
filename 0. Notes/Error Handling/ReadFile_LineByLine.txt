package main

import (
	"bufio"
	"fmt"
	"log"
	"os"
)

func main() {
	f, err := os.Open("file.txt")
	defer f.Close()
	if err != nil {
		log.Fatal(err)
	}

	// creates a scanner to read the file line by line
	s := bufio.NewScanner(f)

	//s.Scan returns false when EOF
	for s.Scan() {
		//.Text returns the line read
		fmt.Println(s.Text())
	}

	if err != nil {
		log.Fatal(err)
	}
}
