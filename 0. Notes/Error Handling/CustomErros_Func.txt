package main

import (
	"errors"
	"fmt"
	"log"
)

func volume(r float64) (float64, error) {
	if r < 0 {
		//errors.New creates a custom error
		return 0, errors.New("Volume calc failed; radius is negative")
	}
	return (4.0 / 3.0) * 3.14 * r * r * r, nil
}

func main() {
	radius := -5.0
	vol, err := volume(radius)
	if err != nil {
		log.Println(err)
		return
	}
	fmt.Println("Volume is: ", vol)
}
