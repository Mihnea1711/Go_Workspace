package main

import "fmt"

func main() {
	// & => memory
	// * => value

	// clasic pointer declaration
	//var <name> *type
	var ptr *int
	fmt.Println(ptr) // returns nil

	val := 123
	ptr = &val
	fmt.Println(ptr)  //returns memory addr
	fmt.Println(*ptr) //returns the value

	// pointer decl using new keyword
	var myPtr *int = new(int)
	fmt.Println(myPtr)  //returns memory addr
	fmt.Println(*myPtr) //returns the 0

	*myPtr = 123
	fmt.Println(*myPtr) //returns the value after assigning

	//declare using the assignment operator (inititalize with an address)
	mySecondPtr := &val
	fmt.Println(mySecondPtr, *mySecondPtr)

	//declare pointer with var keyword	(compiler assigns the address)
	//compiler figures out that this is an address
	var otherPtr = &val
	fmt.Println(otherPtr, *otherPtr)
}
