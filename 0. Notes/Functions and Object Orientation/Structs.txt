package main

import "fmt"

//because Box starts with a capital letter, it is seen as exportable
//best practice is to have a comment line before the struct, like this

// Box is an exportable object
type Box struct {
	depth  float64
	width  float64
	height float64
}

func main() {
	//note that if we don't specify the names of the fields, we need to  put the numbers in order to match the fields! Best practice not to!
	b := Box{depth: 5, width: 5, height: 5}
	fmt.Println(b)
	fmt.Println(b.depth)

	structPtr := &b
	(*structPtr).depth = 8
	fmt.Println(b)

	//go allows us to use ptr to structs without the (*..)
	structPtr.width = 10
	fmt.Println(b)

}
