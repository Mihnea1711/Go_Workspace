package main

import "fmt"

//variadic functions
//parameters vary in length
//when we don't know the contents

// here we are using the pack operator (...int)
func add(values ...int) {
	total := 0
	for _, value := range values {
		total += value
	}
	fmt.Println(values, total)
}

func main() {
	values := []int{1, 2, 3, 4}
	//... is called the unpack operator when calling the variadic func (unpacks the elements of a slice)
	add(values...)

	values = []int{1, 2, 3, 4, 5, 6, 7}
	add(values...)

	//this has the same behaviour
	add(5, 8, 7, 6, 3)
}
