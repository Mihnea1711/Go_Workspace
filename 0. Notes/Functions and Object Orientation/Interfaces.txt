package main

import "fmt"

//working with interfaces

type Food interface {
	Nutrition() string
	FoodType() string
}

type Apple struct{}

func (a Apple) Nutrition() string {
	return "Apples are carb heavy!"
}
func (a Apple) FoodType() string {
	return "Apples are fruits"
}

type Celery struct{}

func (c Celery) Nutrition() string {
	return "Celery is kaki!"
}
func (c Celery) FoodType() string {
	return "Celery is a veggie"
}

func main() {
//we have a slice of Food types (food is the interface)
//basically anything can implement the interface as long it implements its methods
	foods := []Food{Apple{}, Celery{}}
	for _, food := range foods {
		fmt.Println(food.Nutrition(), food.FoodType())
	}
}
